# 19.删除链表的倒数第N个节点

[力扣题目链接](https://leetcode.com/problems/remove-nth-node-from-end-of-list/)

## 題目
给你一个链表，删除链表的倒数第 n 个结点，并且返回链表的头结点。

进阶：你能尝试使用一趟扫描实现吗？

示例 1：

输入：head = [1,2,3,4,5], n = 2
输出：[1,2,3,5]

示例 2：

输入：head = [1], n = 1
输出：[]

示例 3：

输入：head = [1,2], n = 1
输出：[1]

## 程式碼
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} n
 * @return {ListNode}
 */
var removeNthFromEnd = function(head, n){
  // 創建虛擬頭節點並指向head
  let dummy = new ListNode(0,head);
  // 定義快慢指針，並都放在dummy位置
  let fast = dummy;
  let slow = dummy;
  // 先讓fast偷跑n+1格，fast前進一格重複n+1次
  for(let i = 0; i < n+1; i++){
      fast = fast.next // 讓fast前進一格
  }
  // 讓快慢指針一起前進，直到fast移動到null
  while(fast!==null){
      slow = slow.next;
      fast = fast.next;
  }
  // 刪除倒數第N個節點，直接讓slow指向下下個節點
  slow.next= slow.next.next
  // 最後返回dummy.next指向的新head
  return dummy.next;
  
};
