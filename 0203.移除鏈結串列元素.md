# 203.移除鏈結串列元素

[力扣题目链接](https://leetcode.com/problems/remove-linked-list-elements/)

题意：删除链表中等于给定值 val 的所有节点。

示例 1：
输入：head = [1,2,6,3,4,5,6], val = 6
输出：[1,2,3,4,5]

示例 2：
输入：head = [], val = 1
输出：[]

示例 3：
输入：head = [7,7,7,7], val = 7
输出：[]


/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var removeElements = function(head, val) {
    // 建立虛擬頭節點
    let dummy = new ListNode(0);
    // 將虛擬頭節點指向真正的頭節點
    dummy.next = head;
    // 宣告行走指標從dummy開始出發
    let curr = dummy;
    // curr現在是指向虛擬頭節點。所以我們要以curr.next來當條件，並且因為ListNode的結尾一定是指向null，以此當作終止條件
    while(curr.next!=null){
        // 判斷curr.next.val是否等於要刪除的val
        if(curr.next.val===val){
            curr.next = curr.next.next
        }else{
            curr = curr.next
        }
    }
    
    return dummy.next;
    
};
