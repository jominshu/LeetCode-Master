## 題目
给定一个链表，两两交换其中相邻的节点，并返回交换后的链表。

你不能只是单纯的改变节点内部的值，而是需要实际的进行节点交换。

## 思路
兩兩節點交換需要一個虛擬頭節點協助，利用虛擬頭節點在節點兩兩交換之前，先牽住後面所有節點，再進行交換
步驟如下:
1. prev.next=node2
2. node1.next = node2.next;
3. node2.next = node1
4. prev=node1

## 程式碼
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */

/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var swapPairs = function(head) {
    // 建立虛擬頭節點
    const dummy = new ListNode(0);
    // 讓建立的虛擬頭節點指向head頭節點
    dummy.next = head;
    // 讓prev指標指向dummy
    let prev = dummy;
    // 檢查後兩個節點是否還有節點，若為奇數個節點，則檢查下下個節點
    while(prev.next !==null && prev.next.next!==null){
    // 建立頭兩個節點指標
    let node1=prev.next
    let node2=prev.next.next
    
    prev.next=node2 // 把前一節點指向node2
    node1.next=node2.next // 把node1.next也指向node2的下一個節點，牽住後面的所有節點
    node2.next=node1 // 交換:把node2.next接回node1

    prev=node1; // prev推進到交換後的node1節點位置，為下一輪作準備
    }
    // 所有交換完畢後回傳新的頭節點
    return dummy.next;


};
